// ─── HERO CANVAS ────────────────────────────────
const CONNECTIONS = [
  [11,12],[11,13],[13,15],[12,14],[14,16],
  [11,23],[12,24],[23,24],
  [23,25],[25,27],[24,26],[26,28],
  [0,11],[0,12],
];

function genPose(t) {
  const p = t % 1;
  return {
    0:  [320,60],
    11: [300+Math.sin(p*Math.PI*2)*20, 180+Math.cos(p*Math.PI)*30],
    12: [340+Math.sin(p*Math.PI*2)*20, 180+Math.cos(p*Math.PI)*30],
    13: [270+Math.sin(p*Math.PI*2)*30, 230],
    14: [370+Math.sin(p*Math.PI*2)*30, 230],
    15: [250, 290+Math.sin(p*Math.PI*2)*20],
    16: [390, 290+Math.sin(p*Math.PI*2)*20],
    23: [305, 310+Math.sin(p*Math.PI)*10],
    24: [335, 310+Math.sin(p*Math.PI)*10],
    25: [285,390], 26:[355,390],
    27: [270,460], 28:[370,460],
  };
}

let heroT = 0, heroRAF;
function initHeroCanvas() {
  const canvas = document.getElementById('heroCanvas');
  if (!canvas) return;
  const ctx = canvas.getContext('2d');
  canvas.width = canvas.offsetWidth;
  canvas.height = canvas.offsetHeight;

  function draw() {
    ctx.clearRect(0,0,canvas.width,canvas.height);
    const sx = canvas.width/640, sy = canvas.height/520;
    ctx.save(); ctx.scale(sx, sy);

    const pose = genPose(heroT);

    ctx.lineWidth = 3;
    CONNECTIONS.forEach(([a,b]) => {
      if (!pose[a]||!pose[b]) return;
      const err = a===11||a===13||b===11||b===13;
      ctx.strokeStyle = err ? 'rgba(239,68,68,0.85)' : 'rgba(91,141,255,0.75)';
      ctx.beginPath();
      ctx.moveTo(pose[a][0],pose[a][1]);
      ctx.lineTo(pose[b][0],pose[b][1]);
      ctx.stroke();
    });

    Object.entries(pose).forEach(([i,[x,y]]) => {
      const err = i==11||i==13;
      ctx.beginPath(); ctx.arc(x,y,err?7:5,0,Math.PI*2);
      ctx.fillStyle = err ? '#ef4444' : '#5b8dff';
      ctx.fill();
    });

    // Grid
    ctx.strokeStyle = 'rgba(91,141,255,0.06)';
    ctx.lineWidth = 1;
    for(let i=0;i<640;i+=60){
      ctx.beginPath();ctx.moveTo(i,0);ctx.lineTo(i,520);ctx.stroke();
    }
    for(let j=0;j<520;j+=60){
      ctx.beginPath();ctx.moveTo(0,j);ctx.lineTo(640,j);ctx.stroke();
    }

    ctx.restore();
    heroT += 0.003;
    heroRAF = requestAnimationFrame(draw);
  }

  if (heroRAF) cancelAnimationFrame(heroRAF);
  draw();
}
